const ProductCard = ({ product }) => {
  const [isLoading, setIsLoading] = useState(false);
  
  const handleAddToCart = async () => {
    setIsLoading(true);
    try {
      await addToCart(product.id);
      showNotification('Prodotto aggiunto!');
    } catch (error) {
      showError('Errore nell\'aggiunta');
    } finally {
      setIsLoading(false);
    }
  };
  
  return (
    <div className="product-card">
      <img src={product.image} alt={product.name} />
      <h3>{product.name}</h3>
      <p className="price">‚Ç¨{product.price}</p>
      <button 
        onClick={handleAddToCart}
        disabled={isLoading}
        className="add-to-cart-btn"
      >
        {isLoading ? 'Aggiungendo...' : 'Aggiungi al carrello'}
      </button>
    </div>
  );
};

Task Management App

App per la gestione di progetti e task con collaborazione in tempo reale.

TaskBoard.vue 
<template>
  <div class="task-board">
    <div 
      v-for="column in columns" 
      :key="column.id"
      class="task-column"
      @drop="onDrop($event, column.id)"
      @dragover.prevent
    >
      <h3>{{ column.title }}</h3>
      <TaskCard
        v-for="task in column.tasks"
        :key="task.id"
        :task="task"
        @drag-start="onDragStart"
        draggable="true"
      />
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      draggedTask: null,
      columns: [
        { id: 'todo', title: 'Da fare', tasks: [] },
        { id: 'progress', title: 'In corso', tasks: [] },
        { id: 'done', title: 'Completato', tasks: [] }
      ]
    }
  },
  methods: {
    onDragStart(task) {
      this.draggedTask = task;
    },
    onDrop(event, columnId) {
      if (this.draggedTask) {
        this.moveTask(this.draggedTask.id, columnId);
        this.draggedTask = null;
      }
    }
  }
}
</script>
